#### 2.基础的链表声明如下

```js
//创建链表的结构
//相当于c/c++里面的结构体的概念
//泛化,使用对象表示链表每一个结构的构造
const listb ={
		  value:2,
			next : null
}
const lista ={
			value:1,
			next: listb
}
console.log(lista.next.value);
//============下面使用函数式方法快速实现链表结构============
//函数式申明
function Node(value){
		 this.value = value;
		 this.next = null;
}
//创建一些节点操作   
const my1 = new Node(1);
const my2 = new Node(2);
const my3 = new Node(3);
const my4 = new Node(4);
//构造节点之间的联系
my1.next= my2;
my2.next= my3;
my3.next= my4;
my4.next= null;
//输出
console.log(my1.next.value,my2.next.value);

```

#### 3.常规的数组遍历方法

```js
//数组的遍历输出
var test = [1,2,3,4,5,6];
function traverse(arr){
			//判断数组为空情况
			if(arr == null) return;
			for(var i = 0;i < arr.length; i++){
			console.log(arr[i]+ " ");
			}
			console.log("数组打印输出完毕");
}
traverse(test);
```

#### 4.数组递归遍历

```js
//数组的递归遍历操作函数
var test = [1,2,3,4,5,6];
function traverse_one(arr,i){
    if(arr == null || arr.length <= i){
        console.log("数组打印完毕!");
        return;
    }
    console.log(arr[i]);
    traverse_one(arr,i+1);
}
traverse_one(test,0);
```

#### 5.链表的常规遍历 (ps:未知链表的长度可以使用while循环操作)

```js
function traverse_Link(root){
    var temp = root;
    //使用while循环遍历操作
    while(true){
        if(temp != null){
            console.log(temp.value);
        }else{
            console.log("链表打印输出完毕!");
            break;
        }
        temp = temp.next;
    }
}
```

#### 6.链表的递归遍历解法

```js
//链表的递归调用方式
function traverse_Link_one(root){
     if(root == null) {console.log("链表打印完毕!"); return;}
     console.log(root.value);
     traverse_Link_one(root.next); //递归调用方式
}
```
